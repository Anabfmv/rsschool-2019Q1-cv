function showNotification(NameTitle, BodyText) {
  Notification.requestPermission(function(result) {
    if (result === 'granted') {
      navigator.serviceWorker.ready.then(function(registration) {
        registration.showNotification(NameTitle, {
          body: BodyText
        });
      });
    }
  });
}
